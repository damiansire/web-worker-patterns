Servicio de Worker

import { injectable, signal } from '@angular/core';
import { Subject } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class WorkerService {
    private worker: Worker;
    private messageSubject = new Subject<string>();
    
    // Angular 21: Signals para estado reactivo
    message = signal('');
    
    constructor() {
        this.worker = new Worker('./worker.js');
        this.setupWorker();
    }
    
    private setupWorker(): void {
        this.worker.onmessage = (e) => {
            this.messageSubject.next(e.data);
            this.message.set(e.data); // Update signal
        };
        
        this.worker.onerror = (error) => {
            this.messageSubject.error(error);
        };
    }
    
    sendMessage(message: string): void {
        this.worker.postMessage(message);
    }
    
    getMessages() {
        return this.messageSubject.asObservable();
    }
}

import { Component, signal, inject } from '@angular/core';
import { WorkerService } from './worker.service';

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html',
    standalone: true
})
export class AppComponent {
    // Angular 21: inject() function
    private workerService = inject(WorkerService);
    
    // Angular 21: Signals instead of properties
    inputMessage = signal('');
    messages = signal<string[]>([]);
    
    constructor() {
        // Angular 21: toSignal() for converting Observables to Signals
        import { toSignal } from '@angular/core/rxjs-interop';
    }
    
    sendMessage(): void {
        const msg = this.inputMessage();
        if (msg.trim()) {
            this.workerService.sendMessage(msg);
            this.inputMessage.set('');
        }
    }
}
