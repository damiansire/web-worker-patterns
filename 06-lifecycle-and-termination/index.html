Ejemplo 06: Gestión del ciclo de vida de Workers

💡 ¿Qué demuestra este ejemplo? Es importante gestionar correctamente el ciclo de vida de los Workers para liberar recursos y memoria. Este ejemplo muestra cómo crear, usar y terminar Workers de forma apropiada.
📖 Ver Código - ¿Cómo funciona?

1️⃣ Crear Worker (main.js)

let worker = null;

function createWorker() {
    // Crear nueva instancia del worker
    worker = new Worker('worker.js');
    
    // Configurar listeners
    worker.onmessage = handleMessage;
    worker.onerror = handleError;
}

2️⃣ Usar Worker (main.js)

// Enviar tarea al worker
function startTask() {
    if (worker) {
        worker.postMessage({
            action: 'startTask',
            duration: 5000
        });
    }
}

3️⃣ Terminar Worker (main.js)

function terminateWorker() {
    if (worker) {
        // Terminar inmediatamente el worker
        worker.terminate();
        worker = null;
        
        console.log('Worker terminado');
    }
}

4️⃣ Auto-terminación desde Worker (worker.js)

// El worker puede terminarse a sí mismo
self.onmessage = function(e) {
    if (e.data.action === 'finish') {
        // Hacer limpieza
        cleanup();
        
        // Auto-terminar
        self.close();
    }
};
