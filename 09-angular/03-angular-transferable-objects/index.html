Servicio con Transferencia

import { injectable } from '@angular/core';
import { Subject } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class TransferService {
    private worker: Worker;
    private resultSubject = new Subject<any>();
    
    constructor() {
        this.worker = new Worker('./worker.js');
        this.setupWorker();
    }
    
    // M√©todo con transferencia (R√ÅPIDO - transfiere propiedad)
    processWithTransfer(imageData: ImageData) {
        const buffer = imageData.data.buffer;
        
        // Transferir en lugar de clonar
        this.worker.postMessage(
            { imageData },
            [buffer] // Segundo par√°metro: objetos a transferir
        );
        
        // El buffer original ahora est√° vac√≠o
        console.log('Buffer transferido, longitud:', buffer.byteLength); // 0
        
        return this.resultSubject.asObservable();
    }
    
    // M√©todo sin transferencia (LENTO - clona datos)
    processWithClone(imageData: ImageData) {
        // Normal postMessage clona los datos
        this.worker.postMessage({ imageData });
        
        // El buffer original sigue accesible
        console.log('Buffer clonado, longitud:', imageData.data.buffer.byteLength); // Sin cambios
        
        return this.resultSubject.asObservable();
    }
    
    private setupWorker(): void {
        this.worker.onmessage = (e) => {
            this.resultSubject.next(e.data);
        };
    }
}

Componente que usa el Servicio

import { Component, signal, inject } from '@angular/core';

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html',
    standalone: true
})
export class AppComponent {
    // Angular 21: inject() function
    private transferService = inject(TransferService);
    
    // Angular 21: Signals
    useTransfer = signal(true);
    duration = signal(0);
    
    processImage(): void {
        const imageData = this.createTestImage();
        const startTime = performance.now();
        
        const processFn = this.useTransfer()
            ? this.transferService.processWithTransfer(imageData)
            : this.transferService.processWithClone(imageData);
        
        processFn.subscribe(result => {
            this.duration.set(performance.now() - startTime);
        });
    }
}
