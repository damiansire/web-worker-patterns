Servicio con Manejo de Errores

import { Injectable } from '@angular/core';
import { Observable, Subject, throwError } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class ErrorHandlerService {
    private worker: Worker;
    private resultSubject = new Subject<any>();
    private errorSubject = new Subject<any>();
    
    constructor() {
        this.worker = new Worker('./worker.js');
        this.setupWorker();
    }
    
    private setupWorker(): void {
        // Manejar mensajes normales
        this.worker.onmessage = (e) => {
            if (e.data.success) {
                this.resultSubject.next(e.data.result);
            } else {
                // Error manejado por el worker
                this.errorSubject.next(e.data.error);
            }
        };
        
        // Manejar errores no capturados del worker
        this.worker.onerror = (error) => {
            console.error('Worker error:', {
                message: error.message,
                filename: error.filename,
                lineno: error.lineno
            });
            this.errorSubject.next({
                type: 'uncaught',
                message: error.message
            });
        };
    }
    
    processData(data: any): Observable<any> {
        this.worker.postMessage(data);
        return this.resultSubject.asObservable();
    }
    
    getErrors(): Observable<any> {
        return this.errorSubject.asObservable();
    }
}

Componente con Manejo de Errores

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html'
})
export class AppComponent {
    errors: any[] = [];
    results: any[] = [];
    
    constructor(private errorHandlerService: ErrorHandlerService) {
        // Suscribirse a errores
        this.errorHandlerService
            .getErrors()
            .subscribe(error => {
                this.handleError(error);
            });
    }
    
    triggerError(errorType: string): void {
        console.log('Triggeando error:', errorType);
        
        this.errorHandlerService
            .processData({ type: errorType })
            .subscribe({
                next: result => {
                    this.results.push(result);
                },
                error: error => {
                    console.error('Observable error:', error);
                }
            });
    }
    
    private handleError(error: any): void {
        this.errors.push({
            timestamp: new Date(),
            error
        });
        console.error('Error capturado:', error);
    }
}
