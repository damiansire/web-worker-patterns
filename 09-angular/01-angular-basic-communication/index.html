Ejemplo 01: Enviando y recibiendo mensajes con Angular

💡 ¿Qué hace este ejemplo? Este es el "Hola Mundo" de los Web Workers con Angular. Aprende cómo integrar Web Workers en una aplicación Angular usando servicios y observables para una comunicación reactiva.
📖 Ver Código - ¿Cómo funciona?

Servicio de Worker

import { Injectable } from '@angular/core';
import { Observable, Subject } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class WorkerService {
    private worker: Worker;
    private messageSubject = new Subject<string>();
    
    constructor() {
        this.worker = new Worker('./worker.js');
        this.setupWorker();
    }
    
    private setupWorker(): void {
        this.worker.onmessage = (e) => {
            this.messageSubject.next(e.data);
        };
        
        this.worker.onerror = (error) => {
            this.messageSubject.error(error);
        };
    }
    
    sendMessage(message: string): void {
        this.worker.postMessage(message);
    }
    
    getMessages(): Observable<string> {
        return this.messageSubject.asObservable();
    }
}

Componente

import { Component } from '@angular/core';
import { WorkerService } from './worker.service';
import { Subscription } from 'rxjs';

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html'
})
export class AppComponent {
    message: string = '';
    messages: string[] = [];
    private subscription: Subscription;
    
    constructor(private workerService: WorkerService) {
        // Suscribirse a los mensajes del worker
        this.subscription = this.workerService
            .getMessages()
            .subscribe(msg => {
                this.messages.push(msg);
            });
    }
    
    sendMessage(): void {
        if (this.message.trim()) {
            this.workerService.sendMessage(this.message);
            this.message = '';
        }
    }
    
    ngOnDestroy(): void {
        this.subscription.unsubscribe();
    }
}

Template

<!-- Input para el mensaje -->

    Enviar Mensaje
</button>

<!-- Lista de mensajes -->
<div *ngFor="let msg of messages" class="message">
    {{ msg }}
</div>

    const mensaje = e.data;
    console.log('Worker recibió:', mensaje);
    
    // Procesar y responder
    const respuesta = `Procesé tu mensaje: "${mensaje}"`;
    self.postMessage(respuesta);
};

🎯 Ventajas de Angular + Web Workers

  • Servicios: Encapsulación y reutilización del código del worker
  • RxJS: Comunicación reactiva con Observables
  • Dependency Injection: Fácil de testear y mockear
  • TypeScript: Tipado fuerte para mayor seguridad
  • Lifecycle: Gestión automática de subscriptions
