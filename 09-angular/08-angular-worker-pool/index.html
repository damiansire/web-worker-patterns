Servicio de Worker Pool

import { Injectable, OnDestroy } from '@angular/core';
import { BehaviorSubject, Observable } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class WorkerPoolService implements OnDestroy {
    private poolSize: number;
    private workers: WorkerPoolItem[] = [];
    private taskQueue: Task[] = [];
    private completed = 0;
    private active = 0;
    
    stats = new BehaviorSubject<any>({
        poolSize: 0,
        active: 0,
        queued: 0,
        completed: 0
    });
    
    initPool(size: number = 4): void {
        this.poolSize = size;
        this.workers = [];
        
        for (let i = 0; i < size; i++) {
            const worker = new Worker('./worker.js');
            
            const item: WorkerPoolItem = {
                id: i,
                worker,
                busy: false
            };
            
            worker.onmessage = (e) => {
                this.handleWorkerComplete(item, e.data);
            };
            
            this.workers.push(item);
        }
        
        this.updateStats();
    }
    
    addTask(task: Task): void {
        this.taskQueue.push(task);
        this.processQueue();
        this.updateStats();
    }
    
    private processQueue(): void {
        const available = this.workers.filter(w => !w.busy);
        
        for (const worker of available) {
            if (this.taskQueue.length > 0) {
                const task = this.taskQueue.shift();
                worker.busy = true;
                worker.worker.postMessage(task.data);
                this.active++;
            }
        }
    }
    
    private handleWorkerComplete(item: WorkerPoolItem, result: any): void {
        item.busy = false;
        this.active--;
        this.completed++;
        
        // Procesar siguiente tarea
        this.processQueue();
        this.updateStats();
    }
    
    getStats(): Observable<any> {
        return this.stats.asObservable();
    }
    
    private updateStats(): void {
        this.stats.next({
            poolSize: this.poolSize,
            active:  this.active,
            queued: this.taskQueue.length,
            completed: this.completed
        });
    }
    
    ngOnDestroy(): void {
        this.workers.forEach(w => w.worker.terminate());
    }
}

interface WorkerPoolItem {
    id: number;
    worker: Worker;
    busy: boolean;
}

interface Task {
    id: number;
    data: any;
}

Componente usando el Pool

@Component({
    selector: 'app-root'
})
export class AppComponent {
    stats: any = {};
    
    constructor(private poolService: WorkerPoolService) {
        // Inicializar pool con 4 workers
        this.poolService.initPool(4);
        
        this.poolService.getStats().subscribe(stats => {
            this.stats = stats;
        });
    }
    
    addTasks(count: number): void {
        for (let i = 0; i < count; i++) {
            this.poolService.addTask({
                id: i,
                data: { taskId: i, duration: 1000 }
            });
        }
    }
}
