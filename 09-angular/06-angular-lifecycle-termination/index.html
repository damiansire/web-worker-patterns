import { Injectable, OnDestroy } from '@angular/core';

@Injectable({ providedIn: 'root' })
export class LifecycleService implements OnDestroy {
    private worker: Worker | null = null;
    
    createWorker(): boolean {
        if (this.worker) {
            console.warn('Worker ya existe');
            return false;
        }
        
        this.worker = new Worker('./worker.js');
        this.setupWorker();
        return true;
    }
    
    terminateWorker(): boolean {
        if (this.worker) {
            this.worker.terminate();
            this.worker = null;
            return true;
        }
        return false;
    }
    
    private setupWorker(): void {
        if (!this.worker) return;
        
        this.worker.onmessage = (e) => {
            console.log('Mensaje del worker:', e.data);
        };
    }
    
    get workerActive(): boolean {
        return this.worker !== null;
    }
    
    ngOnDestroy(): void {
        // Auto-terminar al destruir el componente
        if (this.worker) {
            this.worker.terminate();
            this.worker = null;
        }
    }
}

@Component({
    selector: 'app-root',
    templateUrl: './app.component.html'
})
export class AppComponent implements OnDestroy {
    isWorkerActive: boolean = false;
    
    constructor(private lifecycleService: LifecycleService) {
        this.isWorkerActive = this.lifecycleService.workerActive;
    }
    
    createWorker(): void {
        this.isWorkerActive = this.lifecycleService.createWorker();
    }
    
    terminateWorker(): void {
        this.isWorkerActive = !this.lifecycleService.terminateWorker();
    }
    
    ngOnDestroy(): void {
        // El servicio auto-terminar√° el worker
        this.lifecycleService.ngOnDestroy();
    }
}
